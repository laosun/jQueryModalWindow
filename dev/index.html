/*

Для истетики уберем большие буквы в параметрах:
containerCSS - это 			container : {}
elementCSS — это 			element : {}
shadowCSS — это будет 		shadow : {}

Так же внутри вышеописанных объектов можно будет вызывать специфичные параметры, которые вычленяются скриптом и не попадают в непосредственный CSS. Пока только несколько параметров: 

container["content"] — контент блока container, вставляется в блок через $.prepend()

element["content"] — контент блока element, вставляется в блок через $.html()
element["top-space"] — высота блока .modal-win-sepparator-top
element["bottom-space"] — высота блока .modal-win-sepparator-bottom

shadow["disable-any"] — boolean. true когда окно появляется — у всех остальных окон тень исчезает, false — тени на остальных окнах остаются, если они были включены. Дефолт true.
shadow["animate-opacity"] — стартовое и конечное значение прозрачности, для анимации тени. Если оно не указано — то все происходит моментально.
shadow["animate-show"] — время появляения тени. Дефолт 500.
shadow["animate-hide"] — время исчезновения тени. Дефолт 500.

Пример вызова:

$.ModalWin({

	id : "hello",

	close

		onshadow		: 		true, 				// ванианты: true/false

		element 		: 		".close-button", 	// передаем имя селектора, который будет искаться в блоке .modal-win, если ничего не задано — не обрабатываем
	
	},

	shadow : {

		"disable-any"		: true,		

		"animate-opacity"	: 0,
		"animate-show"		: 500,
		"animate-hide"		: 500,

		"background-color" 	: "#000",
		"opacity" 			: 0.4

	},

	container : {

		content 	: 	"... тут например кнопка закрытия окна ..."

	},

	element : {

		"background-color" 	: "#fff",

		"top-space" 		: "50px",
		"bottom-space" 		: "40px",

		content		:	"... тут например прелодер"

	},

	load : {
		
		url 		: "/dev/stuff/ex2.html", // опц. загрузчик контента

		append	: true, // Если true — то загружаемый контент будет вставлен через append, если false — то контент будет будет заменять все содержимое блоко. Дефолт false.

		once		: true // Однажды загружаемый контент, если false — то при закрытии окна оно будет удаляться из DOM. Дефолт false.

	},

	call : {
		
		before		: function () {}, 		// опц. функция, которая срабатывает до вызова загрузчика

		after		: function () {}, 		// функция срабатывает после загрузки контента

		always		: function () {}, 		// функция срабатывает после after, но при этом каждый раз при открытии окна, даже если параметр load.once = true

		close		:	function () {} 		// срабатывает до закрытия окна, в ней можно анимировать исчезновение блока element

	}

});

*/

x.loadurl("/dev/stuff/ex2.html");
